from functools import lru_cache


def moves(a):
    return a + 1, a + 5, a * 3


@lru_cache(None)
def f(a):
    if a >= 41:
        return 'END'
    elif any(f(x) == 'END' for x in moves(a)):
        return 'П1'
    elif all(f(x) == 'П1' for x in moves(a)):
        return 'В1'
    elif any(f(x) == 'В1' for x in moves(a)):
        return 'П2'
    elif all(f(x) == 'П2' or f(x) == 'П1' for x in moves(a)):
        return 'В2'


for i in range(1, 41):
    print(i, f(i))
